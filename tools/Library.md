# Library-plugin

# cargo utils
# cargo expand: show macro expansion
cargo install cargo-expand
# ebuild for gentoo
cargo install cargo-ebuild
# alternative grep
cargo install ripgrep
# view file hex
cargo install hexyl xxv
# alternative ls
cargo install exa

cargo install bottom
# benchmarking tool
cargo install hyperfine
# alternative cat
cargo intall bat
