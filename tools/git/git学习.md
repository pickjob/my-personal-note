```bash
# 初始化项目
git init project_name
# 添加到版本库
git add file_name
# 提交
git commit -m message
# 当前状态
git status
# list branch
#   本地分支查看
    git branch
#   远程分支查看
    git branch -r
#   所有分支
    git branch -a
# create branch
git checkout -b my-branch-name [origin/branch-name]
# switch branch
git checkout my-branch-name
# update branch
git pull
# push branch
git push
```
