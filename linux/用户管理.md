# 用户管理
```bash
# 用户信息
    who
    whoami
    id
# 用户的新增、修改、删除
#   useradd / usermod / userdel
    useradd --groups wheel --shell /bin/bash china
# 修改密码
    passwd [USER]
# 组的新增、修改、删除
#   groupadd / groupmod / groupdel
# 用户组的更改
    gpasswd --add USER GROUP
    gpasswd --delete USER GROUP
# 更改当前用户
    su -
    su USER
```
