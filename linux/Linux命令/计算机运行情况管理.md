# 计算机运行情况管理
- free : 显示内存信息
    - -b : 单位为byte
    - -k : 单位为KB
    - -m : 单位为MB
    - -g : 单位为GB
    - -t : 显示RAM + swap 总和
    - -s delay : 设置多少秒刷新一次
    - -c count : 设置刷新次数
- pstree : 树状图显示程序
    - -a : 显示命令行参数
    - -p : 显示进程号
    - -h : 高亮显示当前执行的进程
    - -u : 显示用户名称
- ps : 报告程序状况
    - -ef
    - -axu
    - -ejH : 显示进程树
    - -eLf : 获得线程信息
- pmap : 显示程序的内存信息
    - [pid]
- stat : 显示文件/文件夹的详细信息
    - [file/directory]
- top : 显示进程信息
    - 操作
        - alt + hjkl
- vmstat : 显示虚拟内存的统计信息
- kill : 杀死进程,默认发送SIGTERM信号
    - [-s signal]
    - [-p PID]
    - [-l]
